from scipy.optimize import curve_fit

fig,ax = plt.subplots()
plt.scatter(dfi['sipmEnergy'], dfi['slmEnergy'])

def func(x,m,b):
    return m*x + b 

initial_guess = [1.2, 0.03]
pars, pcov = curve_fit(func, dfi['sipmEnergy'], dfi['slmEnergy'], p0=initial_guess)

xfit = np.linspace(np.amin(dfi['sipmEnergy']),np.amax(dfi['sipmEnergy']))
yfit = func(xfit, pars[0], pars[1])
plt.plot(xfit,yfit,'r-',label=f"Linear Fit\nm={pars[0]:.2f}\nb={pars[1]:.2f}")

plt.xlabel("SLM Energy [???]")
plt.ylabel("SiPM Energy [???]")
plt.title(f"SiPM Energy vs. SLM Energy for Calo/Xtal {calo}/{xtal}")
